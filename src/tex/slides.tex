\begin{frame}{Research question and methods}
  \begin{block}{}
    Which set of $k$ proteins, when blocked simultaneously, most effectively limits the propagation of the disease signal in the endometriosis PPI network; and does the optimal set depend on the chosen measure?
  \end{block}

  \vspace{1em}

  Algorithm types used:
  \begin{itemize}
    \item group centrality measures,
    \item diffusion models.
  \end{itemize}
\end{frame}



\begin{frame}{Data and network construction}
  Source:
  \begin{itemize}
    \item Genes from DisGeNET, UMLS C0014175, score $\ge 0.25$, 166 total
    \item Proteins from STRING, confidence $\ge 0.4$, first-shell expansion
  \end{itemize}

  \vspace{0.5em}

  Results:
  \begin{itemize}
    \item 454 nodes and 21{,}095 edges
    \item Single connected component
    \item Average degree $\approx 93$, density $\approx 0.205$
  \end{itemize}
\end{frame}



\begin{frame}{Null models}
  Random graphs as a sanity check: do the PPI numbers differ from random? 
  All networks are matched on $V = 454$.

  \vspace{0.5em}

  \begin{itemize}
    \item Erd\H{o}s-R\'enyi: random edges at 0.205 density
    \item Barab\'asi-Albert: preferential attachment, $m \approx 53$
    \item Configuration model: preserved degree sequence
  \end{itemize}
\end{frame}

\begin{frame}{Diffusion models}
  Simulate the disease signal spreading. Used to score blocking strategies by how much they reduce propagation.

  \vspace{0.5em}

  \begin{itemize}
    \item SIR, Susceptible-Infected-Recovered: infection rate $\beta$, recovery rate $\gamma$.
    \item IC, Independent Cascade: activation probability $p$.
  \end{itemize}

  \vspace{0.5em}

  Spread reduction $= 1 - \dfrac{\text{spread for the blocked network}}{\text{spread for the unblocked network}}$
\end{frame}

\begin{frame}{Group-centrality algorithms}
  Algorithms that pick proteins to block: two structural and one diffusion-aware.

  \vspace{0.5em}

  \begin{itemize}
    \item \texttt{gdeg} - greedy group degree: each step add the node whose neighborhood covers the most uncovered nodes
    \item \texttt{gbtw} - greedy iterative betweenness: each step remove the current highest-betweenness node
    \item CELF - Cost-Effective Lazy Forward: each step pick the node whose blocking best reduces simulated spread
  \end{itemize}

  \vspace{0.5em}

  CELF runs twice, once under SIR and once under IC.
\end{frame}

\begin{frame}{Conclusion and limitations}
  The answer to the research question:

  \begin{itemize}
    \item Best $k$-set: prefix of the \texttt{gbtw} chain, saturates near $k = 200$
    \item Strongly algorithm-dependent: 93\% of top-20 picks are algorithm-specific
  \end{itemize}

  \vspace{0.5em}

  At a realistic small $k$, partial blocking achieves a very modest reduction.

  \vspace{0.5em}

  Main limitations: CELF inner replicates 25 not 50; CELF was not run on nulls; seed cutoff was fixed; single diffusion operating point.
\end{frame}
